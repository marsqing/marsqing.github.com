		<h3>Lookahead</h3>
		<pre><code spellcheck="false" class="ruby">
rule name
	( expr (<span class="fragment highlight-red">comma expr</span>)* ) {
	}
end
		</code></pre>
		<pre><code spellcheck="false" class="ruby">
rule name
	( single_name | multi_name ) {
	}
end
rule single_name
	expr !comma
end
rule multi_names
	expr comma name
end
rule expr
	space [a-z]+
end
		</code></pre>
	</section>

	One rule for each precedence level
	<pre><code spellcheck="false" class="ruby">
rule expr
    mult ('+' mult)*
end

rule multi
    atom ('*' atom)*
end

rule atom
    [0-9]+
end
	</code></pre>
